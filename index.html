function FindProxyForURL(url, host) { url = url.toLowerCase(); host = host.toLowerCase(); isLocal = isInNet(host, "10.0.0.0", "255.0.0.0") || isInNet(host, "172.16.0.0", "255.240.0.0") || isInNet(host, "192.168.0.0", "255.255.0.0") || isInNet(host, "127.0.0.0", "255.255.255.0"); if (isLocal) return "DIRECT"; isYandex = dnsDomainIs(host, ".yandex.ru") || shExpMatch(host, "*.yandex.*"); is2ip = shExpMatch(host, "*.2ip.*"); if (isYandex || is2ip) { return "SOCKS 88.212.235.223:3128"; } return "DIRECT"; }
